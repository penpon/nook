ハッシュタグ抽出

function extractNoteHashtag(feedInfo) {
  // "#Anthropicタグ" → "Anthropic"
  const hashtagMatch = feedInfo.match(
    /^#(.+)タグ$/
  );
  return hashtagMatch ? 
    hashtagMatch[1] : feedInfo;
}
