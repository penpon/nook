タグ抽出ロジック

function extractQiitaTag(feedInfo) {
  // "ChatGPTタグが付けられた新着記事"
  // → "ChatGPT"を抽出
  const tagMatch = feedInfo.match(
    /^(.+)タグが付けられた/
  );
  return tagMatch ? tagMatch[1] : feedInfo;
}
